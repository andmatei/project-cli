```python
from typing import NoReturn

def fail(self, message: str, *, code: str | None = None, exit_code: int = 1) -> NoReturn:
    """Emit an error and exit. Equivalent to: out.error(...); raise typer.Exit(exit_code)."""
    self.error(message, code=code)
    raise typer.Exit(code=exit_code)
```

Add `import typer` at the top if not already present (it shouldn't be — this is a new dependency on typer in `output.py`, so prefer `import typer` lazily inside `fail()` to keep `output.py` typer-free at module load).

```python
def fail(self, message: str, *, code: str | None = None, exit_code: int = 1) -> NoReturn:
    import typer
    self.error(message, code=code)
    raise typer.Exit(code=exit_code)
```

```python
from contextlib import contextmanager
from collections.abc import Iterator

@contextmanager
def edit_milestones(scope) -> Iterator[MilestonesManifest]:
    """Load → mutate → save. Yields the manifest; saves on clean exit."""
    path = scope.milestones_manifest_path
    manifest = load_milestones_manifest(path)
    yield manifest
    path.parent.mkdir(parents=True, exist_ok=True)
    save_milestones_manifest(path, manifest)
```

The `scope` parameter is typed loosely (don't import `Scope` from `workspace` to avoid circular import). Use a Protocol or a structural duck-type. Or just type as `Any`:

```python
@contextmanager
def edit_milestones(scope: "Scope") -> Iterator[MilestonesManifest]:
```

The TYPE_CHECKING-guarded forward reference avoids the circular import.

```python
new_milestone = Milestone(...)
```

```python
def with_provider(
    scope: "Scope",
    *,
    no_push: bool,
) -> "TicketProvider | None":
    """Resolve the project's configured ticketing provider, or None.

    Returns None if --no-push was passed, no ticketing config, or plugin not installed.
    """
    if no_push:
        return None
    from keel.manifest import load_project_manifest
    from keel.workspace import manifest_path
    proj_manifest = load_project_manifest(manifest_path(scope.project))
    return get_provider_for_project(proj_manifest)


def safe_push(out: "Output", op: str, fn) -> None:
    """Call fn(); on failure, log a warning. Used for ticketing side-effects.

    `op` is a short label like "create_milestone" used in the warning message.
    """
    try:
        fn()
    except Exception as e:  # noqa: BLE001
        out.info(f"[warning] ticket {op} failed: {e}")
```
