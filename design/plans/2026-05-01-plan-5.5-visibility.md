## Task 2: `keel task next` — print the next ready task

**Files:**
- Create: `src/keel/commands/task/next.py`
- Modify: `src/keel/commands/task/__init__.py`

Goal: `keel task next [-m <milestone>]` prints the topologically-first ready task. Optional `--start` flag to start it immediately. Optional `--json` for scripting.

- [ ] **Step 1: Implement `cmd_next`**

```python
"""`keel task next` — print or start the next ready task."""

from __future__ import annotations

import typer

from keel.api import (
    Output,
    ErrorCode,
    load_milestones_manifest,
    ready_tasks,
    resolve_cli_scope,
    topological_sort,
)


def cmd_next(
    ctx: typer.Context,
    deliverable: str | None = typer.Option(
        None, "-D", "--deliverable",
        help="Scope: a deliverable instead of the project. Auto-detected from CWD.",
    ),
    project: str | None = typer.Option(
        None, "--project", "-p",
        help="Project name. Auto-detected from CWD if omitted.",
    ),
    milestone: str | None = typer.Option(
        None, "--milestone", "-m", help="Limit to one milestone's tasks.",
    ),
    start: bool = typer.Option(
        False, "--start", help="Start the next task (planned -> active) and record its branch.",
    ),
    json_mode: bool = typer.Option(False, "--json", help="Emit machine-readable JSON to stdout."),
) -> None:
    """Print the topologically-first ready task; with --start, mark it active."""
    out = Output.from_context(ctx, json_mode=json_mode)

    scope = resolve_cli_scope(project, deliverable, out=out)
    manifest = load_milestones_manifest(scope.milestones_manifest_path)

    ready_ids = {t.id for t in ready_tasks(manifest)}
    candidates = [t for t in topological_sort(manifest) if t.id in ready_ids]
    if milestone:
        candidates = [t for t in candidates if t.milestone == milestone]

    if not candidates:
        out.fail("no ready tasks", code=ErrorCode.NOT_FOUND)

    next_task = candidates[0]

    if start:
        # Delegate to existing cmd_start to keep behavior consistent
        from keel.commands.task.start import cmd_start
        ctx.invoke(cmd_start, id=next_task.id, deliverable=deliverable, project=project, branch=None, json_mode=json_mode)
        return

    out.result(
        next_task.model_dump(),
        human_text=f"Next: {next_task.id} ({next_task.milestone}) — {next_task.title}",
    )
```

If `ctx.invoke(cmd_start, ...)` is awkward, just print the task and tell the user to run `keel task start <id>` themselves. Pick whichever is cleaner.

## Task 4: `keel restore` — undo archive

**Files:**
- Create: `src/keel/commands/restore.py`
- Modify: `src/keel/app.py`
- Create: `tests/commands/test_restore.py`

Current state: `keel archive <name>` moves a project to `<projects-dir>/.archive/`. There's no way back without manual `mv`. README markets archive as soft-delete; the restore round-trip should exist.

- [ ] **Step 1: Implement**

```python
"""`keel restore <name>` — bring a project back from .archive/."""

from __future__ import annotations

from pathlib import Path

import typer

from keel.api import ErrorCode, Output, projects_dir


def cmd_restore(
    ctx: typer.Context,
    name: str = typer.Argument(..., help="Project name to restore from .archive/."),
    yes: bool = typer.Option(False, "-y", "--yes", help="Skip confirm prompt."),
    dry_run: bool = typer.Option(False, "--dry-run", help="Print intended operations and exit."),
    json_mode: bool = typer.Option(False, "--json", help="Emit machine-readable JSON to stdout."),
) -> None:
    """Move a project from `<projects-dir>/.archive/` back to the active workspace."""
    out = Output.from_context(ctx, json_mode=json_mode)

    pdir = projects_dir()
    archive_dir = pdir / ".archive" / name
    target_dir = pdir / name

    if not archive_dir.is_dir():
        out.fail(
            f"no archived project named '{name}' (looked in {pdir / '.archive'})",
            code=ErrorCode.NOT_FOUND,
        )
    if target_dir.exists():
        out.fail(
            f"cannot restore: {target_dir} already exists",
            code=ErrorCode.EXISTS,
        )

    if dry_run:
        from keel.api import OpLog
        log = OpLog()
        log.modify_file(target_dir)  # OR add a dedicated rename op if available
        out.info(log.format_summary())
        return

    from keel.prompts import confirm_destructive
    confirm_destructive(f"Restore project '{name}' from archive?", yes=yes)

    archive_dir.rename(target_dir)
    out.result(
        {"restored": name, "path": str(target_dir)},
        human_text=f"Restored: {name} → {target_dir}",
    )
```

NOTE: This does NOT re-init worktrees. Users with code worktrees in their archived project will need to re-run `keel code init` after restore. Document this in the human_text or a follow-up `out.info`.
